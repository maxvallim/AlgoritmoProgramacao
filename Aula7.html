Operadores

// Operadores Aritméticos
int soma = 5 + 3;      // Adição
int subtracao = 5 - 3; // Subtração
int multiplicacao = 5 * 3; // Multiplicação
int divisao = 5 / 3;   // Divisão
int resto = 5 % 3;     // Módulo
// Operadores de Comparação
bool igual = (5 == 3);     // Igual a
bool diferente = (5 != 3); // Diferente de
bool maior = (5 > 3);      // Maior que
bool menor = (5 < 3);      // Menor que
bool maiorIgual = (5 >= 3); // Maior ou igual a
bool menorIgual = (5 <= 3); // Menor ou igual a
// Operadores Lógicos
bool e = (5 > 3 && 3 > 1); // E lógico
bool ou = (5 > 3 || 3 < 1); // Ou lógico
bool nao = !(5 > 3);       // Não lógico
