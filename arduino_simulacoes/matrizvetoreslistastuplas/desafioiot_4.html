Tutorial: Processamento de Placas de Veículos

Explicação das Bibliotecas

  • #include <iostream>: Biblioteca para entrada e saída de dados. Permite exibir mensagens no terminal com cout e ler entradas com cin.
  • #include <string>: Permite manipular textos com facilidade usando o tipo std::string. É usada para armazenar e comparar placas de veículos.

Código do Programa

#include <iostream>
#include <string>
using namespace std;

int main() {
    // Banco de dados de placas cadastradas (matriz 2D)
    string placasCadastradas[5][2] = {
        {"ABC1234", "João"},
        {"XYZ9876", "Maria"},
        {"LMN4567", "Carlos"},
        {"DEF5678", "Ana"},
        {"GHI8901", "Paulo"}
    };

    // Número fixo de placas cadastradas
    int totalPlacasCadastradas = 5;

    // Lista de placas lidas pela câmera
    string placasLidas[] = {"ABC1234", "LMN4567", "ZZZ9999", "XYZ9876", "DEF5678"};

    // Número fixo de placas lidas
    int totalPlacasLidas = 5;

    // Contadores de acesso
    int acessosLiberados = 0;
    int acessosNegados = 0;

    // Processar cada placa lida
    for (int i = 0; i < totalPlacasLidas; i++) {
        string placaAtual = placasLidas[i];
        bool encontrado = false;

        // Verificar se a placa está no banco de dados
        for (int j = 0; j < totalPlacasCadastradas; j++) {
            if (placaAtual == placasCadastradas[j][0]) {
                encontrado = true;
                cout << "Placa: " << placaAtual << " (Proprietário: " << placasCadastradas[j][1] << ") -> Acesso Liberado" << endl;
                acessosLiberados++;
                break;
            }
        }

        // Caso não esteja cadastrada
        if (!encontrado) {
            cout << "Placa: " << placaAtual << " -> Acesso Negado" << endl;
            acessosNegados++;
        }
    }

    // Exibir resumo final
    cout << endl;
    cout << "Resumo:" << endl;
    cout << "Total de acessos liberados: " << acessosLiberados << endl;
    cout << "Total de acessos negados: " << acessosNegados << endl;

    return 0;
}
