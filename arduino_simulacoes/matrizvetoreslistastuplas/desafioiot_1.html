Tutorial: Sensores de Temperatura com std::vector

Explicação das Bibliotecas

  • #include <iostream>: Biblioteca para entrada e saída de dados, usada para exibir mensagens no terminal com cout.
  • #include <cstdlib>: Oferece funções gerais, como rand(), usada para gerar números aleatórios.
  • #include <ctime>: Fornece funcionalidades relacionadas ao tempo, como time(), que inicializa a semente para geração de números aleatórios.
  • #include <vector>: Biblioteca que implementa o tipo std::vector, um container dinâmico que facilita operações como adicionar, remover e obter o número de elementos com o método .size().
  • #include <string>: Permite manipular textos com facilidade usando o tipo std::string.

Código do Programa

#include <iostream>
#include <cstdlib>
#include <ctime>
#include <vector>
#include <string>
using namespace std;

// Função para gerar temperaturas aleatórias
int gerarTemperatura() {
    return rand() % 21 + 15; // Gera números entre 15 e 35
}

void verificarTemperatura(int temperatura, const string& comodo) {
    cout << "Cômodo: " << comodo << ", Temperatura: " << temperatura << "°C -> ";
    if (temperatura > 30) {
        cout << "Status: Ar-condicionado ativado" << endl;
    } else if (temperatura < 18) {
        cout << "Status: Aquecedor ativado" << endl;
    } else {
        cout << "Status: Confortável" << endl;
    }
}

int main() {
    srand(time(0)); // Inicializa a semente para geração de números aleatórios

    vector<string> comodos = {"Sala", "Cozinha", "Quarto", "Banheiro", "Escritório"};

    // Usamos .size() para obter o número de elementos no vetor
    for (size_t i = 0; i < comodos.size(); i++) {
        int temperatura = gerarTemperatura();
        verificarTemperatura(temperatura, comodos[i]);
    }

    return 0;
}
