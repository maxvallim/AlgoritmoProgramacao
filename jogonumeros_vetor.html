#include <iostream>
#include <cstdlib>
#include <ctime>
#include <vector>
#include <string>
using namespace std;

int main()
{
    srand(time(0));

// escolha do nivel
    int nivel;
    cout << "Digite o nivel de dificuldade: ";
    cin >> nivel;

// carga aleatorios
    int numale[nivel];
    for (int i = 0; i < nivel; i++) {
        numale[i]= rand() % 50 + 1;
//        cout << "numero gerado: " << numale[i] << endl;
    }

    string resultado;
    int numlances[nivel];
    int flag=0;
    while (flag == 0) {
        resultado="";
        for (int i = 0; i < nivel; i++) {
            cout << "Digite o numero " << i+1 <<":";
            cin >> numlances[i];
            if (numlances[i] < numale[i]) {
                resultado=resultado + "numero e maior";
            }
            if (numlances[i] > numale[i]) {
                resultado=resultado + "numero e menor";
            }
        }

        if (resultado==""){
            flag=2;
        }
        else {
            cout << resultado << endl;
            string opcao="S";
            cout << "Deseja continuar tentando (S/N)? ";
            cin >> opcao;
            if (opcao=="N") {
                flag=1;
            }
        }
    }

    switch (flag) {
        case 1:
            cout << "Que pena, voce desistiu";
            break;
        case 2:
            cout << "Parabens, voce venceu";
            break;
    }
    return 0;
