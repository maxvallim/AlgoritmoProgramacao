Bibliotecas Incluídas em C++

1.

Biblioteca para operações de entrada e saída padrão.

#include <iostream>

int main() {
    std::cout << "Hello, World!" << std::endl;
    return 0;
}

2.

Biblioteca para o uso de vetores (arrays dinâmicos).

#include <vector>

int main() {
    std::vector<int> numeros = {1, 2, 3, 4, 5};
    for (int num : numeros) {
        std::cout << num << " ";
    }
    return 0;
}

3.

Biblioteca para manipulação de cadeias de caracteres.

#include <string>

int main() {
    std::string saudacao = "Olá, Mundo!";
    std::cout << saudacao << std::endl;
    return 0;
}

4.

Biblioteca para algoritmos padrão, como ordenação e busca.

#include <algorithm>
#include <vector>

int main() {
    std::vector<int> numeros = {5, 3, 1, 4, 2};
    std::sort(numeros.begin(), numeros.end());
    for (int num : numeros) {
        std::cout << num << " ";
    }
    return 0;
}

5.

Biblioteca para o uso de mapas (associações chave-valor).

#include <map>

int main() {
    std::map<std::string, int> idade;
    idade["João"] = 25;
    idade["Maria"] = 30;
    for (const auto& par : idade) {
        std::cout << par.first << ": " << par.second << std::endl;
    }
    return 0;
}

6.

Biblioteca para o uso de conjuntos (coleções de elementos únicos).

#include <set>

int main() {
    std::set<int> numeros = {1, 2, 3, 4, 5};
    numeros.insert(3); // Não será adicionado, pois já existe
    for (int num : numeros) {
        std::cout << num << " ";
    }
    return 0;
}

7.

Biblioteca para funções matemáticas.

#include <cmath>

int main() {
    double raiz = std::sqrt(16.0);
    std::cout << "Raiz quadrada de 16: " << raiz << std::endl;
    return 0;
}

8.

Biblioteca para manipulação de tempo e data.

#include <ctime>

int main() {
    std::time_t agora = std::time(0);
    std::cout << "Hora atual: " << std::ctime(&agora);
    return 0;
}

9.

Biblioteca para operações de leitura e escrita em arquivos.

#include <fstream>

int main() {
    std::ofstream arquivo("exemplo.txt");
    arquivo << "Olá, Arquivo!" << std::endl;
    arquivo.close();
    return 0;
}

10.

Biblioteca para manipulação de threads.

#include <thread>

void funcao() {
    std::cout << "Thread em execução" << std::endl;
}

int main() {
    std::thread t(funcao);
    t.join();
    return 0;
}
